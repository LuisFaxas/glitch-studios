    ```tsx
    "use client"

    import { useMemo } from "react"
    import { X } from "lucide-react"
    import * as Popover from "@base-ui/react/popover"
    import { Slider } from "@/components/ui/slider"

    import type { FilterState, FilterCorpusBounds } from "@/lib/tech/leaderboard"
    // (Use the EXACT same types currently imported by this file. If the existing types live elsewhere —
    //  e.g., a sibling types file — keep the existing imports. Do not invent new types.)

    export interface LeaderboardFiltersProps {
      state: FilterState
      onChange: (next: Partial<FilterState>) => void
      onReset: () => void
      bounds: FilterCorpusBounds
      layout?: "bar" | "vertical"
    }

    export function LeaderboardFilters({
      state,
      onChange,
      onReset,
      bounds,
      layout = "bar",
    }: LeaderboardFiltersProps) {
      const activeCount = useMemo(() => countActiveFilters(state, bounds), [state, bounds])

      const containerClass =
        layout === "bar"
          ? "mb-6 border border-[#222] bg-[#0a0a0a] px-4 py-3"
          : "space-y-6"
      const innerClass =
        layout === "bar"
          ? "flex flex-wrap items-center gap-x-4 gap-y-3"
          : "flex flex-col gap-6"

      return (
        <div data-leaderboard-filters data-layout={layout} className={containerClass}>
          <div className={innerClass}>
            <PriceFilterPopover state={state} bounds={bounds} onChange={onChange} />

            <FilterChipGroup
              label="Year"
              selected={state.years ?? []}
              chips={bounds.years.map((y) => ({ value: y, label: String(y) }))}
              onToggle={(v) => onChange({ years: toggleArray(state.years ?? [], v) })}
            />
            <FilterChipGroup
              label="CPU"
              selected={state.cpus ?? []}
              chips={bounds.cpus.map((c) => ({ value: c.value, label: c.label }))}
              onToggle={(v) => onChange({ cpus: toggleArray(state.cpus ?? [], v) })}
            />
            <FilterChipGroup
              label="RAM"
              selected={state.ram ?? []}
              chips={bounds.ramOptions.map((r) => ({ value: r.value, label: r.label }))}
              onToggle={(v) => onChange({ ram: toggleArray(state.ram ?? [], v) })}
            />
            <FilterChipGroup
              label="Storage"
              selected={state.storage ?? []}
              chips={bounds.storageOptions.map((s) => ({ value: s.value, label: s.label }))}
              onToggle={(v) => onChange({ storage: toggleArray(state.storage ?? [], v) })}
            />
            <FilterChipGroup
              label="Medal"
              selected={state.medals ?? []}
              chips={bounds.medals.map((m) => ({ value: m, label: m }))}
              onToggle={(v) => onChange({ medals: toggleArray(state.medals ?? [], v) })}
            />
            {bounds.subCategories.length > 0 && (
              <FilterChipGroup
                label="Sub-cat"
                selected={state.subCategories ?? []}
                chips={bounds.subCategories.map((sc) => ({ value: sc.slug, label: sc.name }))}
                onToggle={(v) => onChange({ subCategories: toggleArray(state.subCategories ?? [], v) })}
              />
            )}

            <button
              type="button"
              data-reset-filters
              onClick={onReset}
              className={
                layout === "bar"
                  ? "ml-auto inline-flex items-center gap-1 font-mono text-[10px] uppercase tracking-wider text-[#888] hover:text-[#f5f5f0]"
                  : "inline-flex items-center gap-1 font-mono text-[10px] uppercase tracking-wider text-[#888] hover:text-[#f5f5f0]"
              }
            >
              <X className="h-3 w-3" aria-hidden />
              Reset filters{activeCount > 0 ? ` (${activeCount})` : ""}
            </button>
          </div>
        </div>
      )
    }

    // ---- Internal helpers ----

    function countActiveFilters(state: FilterState, bounds: FilterCorpusBounds): number {
      let n = 0
      if (state.minPrice != null || state.maxPrice != null) n += 1
      if ((state.years ?? []).length) n += 1
      if ((state.cpus ?? []).length) n += 1
      if ((state.ram ?? []).length) n += 1
      if ((state.storage ?? []).length) n += 1
      if ((state.medals ?? []).length) n += 1
      if ((state.subCategories ?? []).length) n += 1
      return n
    }

    function toggleArray<T>(arr: T[], value: T): T[] {
      return arr.includes(value) ? arr.filter((x) => x !== value) : [...arr, value]
    }

    interface FilterChipGroupProps {
      label: string
      selected: Array<string | number>
      chips: Array<{ value: string | number; label: string }>
      onToggle: (value: string | number) => void
    }
    function FilterChipGroup({ label, selected, chips, onToggle }: FilterChipGroupProps) {
      const activeCount = chips.filter((c) => selected.includes(c.value)).length
      const labelText = activeCount > 0 ? `${label} (${activeCount})` : label
      return (
        <div data-chip-group={label} className="flex flex-wrap items-center gap-1.5">
          <span className="mr-1 font-mono text-[10px] uppercase tracking-wider text-[#888]">
            {labelText}
          </span>
          {chips.map((c) => {
            const isOn = selected.includes(c.value)
            return (
              <button
                key={String(c.value)}
                type="button"
                onClick={() => onToggle(c.value)}
                aria-pressed={isOn}
                className={
                  "border px-2 py-1 font-mono text-[10px] uppercase tracking-wider transition-colors " +
                  (isOn
                    ? "border-[#f5f5f0] bg-[#1a1a1a] text-[#f5f5f0]"
                    : "border-[#222] text-[#888] hover:border-[#444]")
                }
              >
                {c.label}
              </button>
            )
          })}
        </div>
      )
    }

    interface PriceFilterPopoverProps {
      state: FilterState
      bounds: FilterCorpusBounds
      onChange: (next: Partial<FilterState>) => void
    }
    function PriceFilterPopover({ state, bounds, onChange }: PriceFilterPopoverProps) {
      const min = state.minPrice ?? bounds.priceMin
      const max = state.maxPrice ?? bounds.priceMax
      const isActive = state.minPrice != null || state.maxPrice != null
      const labelText = isActive
        ? `Price: $${min.toLocaleString()}–$${max.toLocaleString()}`
        : "Price"
      return (
        <Popover.Root>
          <Popover.Trigger
            data-price-popover-trigger
            className={
              "border px-3 py-1 font-mono text-[10px] uppercase tracking-wide " +
              (isActive
                ? "border-[#f5f5f0] bg-[#1a1a1a] text-[#f5f5f0]"
                : "border-[#222] text-[#888] hover:border-[#444]")
            }
          >
            {labelText}
          </Popover.Trigger>
          <Popover.Portal>
            <Popover.Positioner sideOffset={8}>
              <Popover.Popup className="border border-[#222] bg-[#0a0a0a] p-4 w-[320px]">
                <Slider
                  value={[min, max]}
                  min={bounds.priceMin}
                  max={bounds.priceMax}
                  step={50}
                  onValueChange={(v) => {
                    const arr = Array.isArray(v) ? Array.from(v) : [v]
                    onChange({ minPrice: arr[0] ?? null, maxPrice: arr[1] ?? null })
                  }}
                />
                <div className="mt-2 flex justify-between font-mono text-[10px] text-[#888]">
                  <span>${min.toLocaleString()}</span>
                  <span>${max.toLocaleString()}</span>
                </div>
              </Popover.Popup>
            </Popover.Positioner>
          </Popover.Portal>
        </Popover.Root>
      )
    }
    ```
