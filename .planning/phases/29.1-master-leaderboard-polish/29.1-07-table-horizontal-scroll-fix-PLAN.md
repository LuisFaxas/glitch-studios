    ```ts
    import { test, expect } from "@playwright/test"

    test.describe("29.1-07 — Desktop table horizontal-scroll fix (D-15/D-16)", () => {
      test.setTimeout(30_000)

      test("@1440: overflow engages on the table wrapper (scrollWidth > clientWidth)", async ({ page }) => {
        await page.setViewportSize({ width: 1440, height: 900 })
        await page.goto("/tech/rankings/laptops")
        const wrapper = page.locator('[data-leaderboard-table]')
        await expect(wrapper).toBeVisible({ timeout: 5_000 })
        const dims = await wrapper.evaluate((el) => ({
          scrollWidth: el.scrollWidth,
          clientWidth: el.clientWidth,
        }))
        expect(dims.scrollWidth).toBeGreaterThan(dims.clientWidth)
        // Confirm the table itself is at least 1600px wide.
        const tableWidth = await page
          .locator('[data-leaderboard-table] table')
          .evaluate((el) => (el as HTMLElement).getBoundingClientRect().width)
        expect(tableWidth).toBeGreaterThanOrEqual(1600)
      })

      test("@1920: table fits inline without overflow", async ({ page }) => {
        await page.setViewportSize({ width: 1920, height: 1080 })
        await page.goto("/tech/rankings/laptops")
        const wrapper = page.locator('[data-leaderboard-table]')
        await expect(wrapper).toBeVisible({ timeout: 5_000 })
        const dims = await wrapper.evaluate((el) => ({
          scrollWidth: el.scrollWidth,
          clientWidth: el.clientWidth,
        }))
        // At 1920 viewport with the page max-w-[1600px] cap, the wrapper width
        // should be ~1600 (or whatever the wrapper computes inside the page chrome).
        // The relevant invariant: the wrapper does NOT need to scroll horizontally.
        expect(dims.scrollWidth).toBeLessThanOrEqual(dims.clientWidth + 1) // +1 tolerates rounding
      })

      test("@1440: scroll the wrapper right; sticky # and Product cells remain in viewport", async ({ page }) => {
        await page.setViewportSize({ width: 1440, height: 900 })
        await page.goto("/tech/rankings/laptops")
        const wrapper = page.locator('[data-leaderboard-table]')
        await expect(wrapper).toBeVisible({ timeout: 5_000 })

        // Scroll the wrapper to its rightmost edge.
        await wrapper.evaluate((el) => {
          el.scrollLeft = el.scrollWidth
        })
        // Wait a tick for any layout to settle.
        await page.waitForTimeout(150)

        // The sticky `#` header cell should still be in viewport.
        const rankHeader = page.locator(
          '[data-leaderboard-table] thead th',
        ).first()
        await expect(rankHeader).toBeInViewport()

        // The sticky Product header cell — typically the second <th>.
        const productHeader = page.locator(
          '[data-leaderboard-table] thead th',
        ).nth(1)
        await expect(productHeader).toBeInViewport()
      })
    })
    ```
