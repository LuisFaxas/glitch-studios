    ```ts
    import { test, expect } from "@playwright/test"

    test.describe("29.1-04 — GlitchTech nav order (D-01)", () => {
      test.setTimeout(30_000)

      const EXPECTED_DESKTOP_ORDER = [
        "Reviews",
        "Rankings",
        "Categories",
        "Compare",
        "Benchmarks",
        "Blog",
        "About",
      ]

      test("desktop tile-nav renders items in D-01 order", async ({ page }) => {
        await page.setViewportSize({ width: 1440, height: 900 })
        await page.goto("/tech")
        // Wait for the nav to hydrate.
        await expect(page.locator("nav a").first()).toBeVisible({ timeout: 5_000 })

        // Filter to tech-nav links (nav anchors whose href starts with /tech).
        const techNavLinks = page.locator('nav a[href^="/tech"]')
        const labels = await techNavLinks.allInnerTexts()
        // Anchors may include extra whitespace and the icon — normalize to the
        // text content trimmed.
        const trimmed = labels.map((s) => s.trim()).filter(Boolean)

        // Find the indices of expected labels and assert ascending order.
        const indices = EXPECTED_DESKTOP_ORDER.map((label) =>
          trimmed.findIndex((t) => t === label || t.startsWith(label)),
        )
        // None should be -1.
        for (let i = 0; i < EXPECTED_DESKTOP_ORDER.length; i++) {
          expect(
            indices[i],
            `Expected to find "${EXPECTED_DESKTOP_ORDER[i]}" in tile-nav`,
          ).toBeGreaterThanOrEqual(0)
        }
        // Indices must be monotonically increasing.
        for (let i = 1; i < indices.length; i++) {
          expect(
            indices[i],
            `Expected "${EXPECTED_DESKTOP_ORDER[i]}" to come after "${EXPECTED_DESKTOP_ORDER[i - 1]}"`,
          ).toBeGreaterThan(indices[i - 1])
        }
      })

      test("mobile overlay menu shows Rankings first, About last", async ({ page }) => {
        await page.setViewportSize({ width: 375, height: 812 })
        await page.goto("/tech")
        // Open the overlay menu — typically a button labelled "Menu" in the bottom tab bar.
        const menuTrigger = page.getByRole("button", { name: /^Menu$/i })
        await expect(menuTrigger).toBeVisible({ timeout: 5_000 })
        await menuTrigger.click()

        // Wait for an overlay item to appear; assert Rankings appears first and About last.
        const overlayItems = page.locator(
          '[role="dialog"] a[href^="/tech"], [data-mobile-overlay] a[href^="/tech"]',
        )
        await expect(overlayItems.first()).toBeVisible({ timeout: 3_000 })

        const labels = (await overlayItems.allInnerTexts()).map((s) => s.trim())
        // First item should be Rankings; last should be About.
        expect(labels[0]).toContain("Rankings")
        expect(labels[labels.length - 1]).toContain("About")
      })
    })
    ```

    Notes:

    - The bottom-tab Menu button label "Menu" is also a guess; if the actual label is different (e.g., "More" or just an icon button with `aria-label="Menu"`), use the correct accessible name.
