**Create `src/components/layout/mobile-content-wrapper.tsx`:**

This is a thin client component wrapper that provides dynamic bottom padding based on player state. The layout.tsx remains a server component.

```typescript
"use client"

import { useAudioPlayer } from "@/components/player/audio-player-provider"
import clsx from "clsx"

export function MobileContentWrapper({ children, className }: { children: React.ReactNode; className?: string }) {
  const { currentBeat, isMinimized } = useAudioPlayer()
  // Player is visually active when a beat is loaded AND not minimized
  const playerActive = !!currentBeat && !isMinimized

  return (
    <main
      className={clsx(
        "page-content flex-1 min-w-0 min-h-screen overflow-x-hidden",
        "transition-[padding-bottom] duration-200 ease-out",
        "md:pb-0",
        playerActive
          ? "pb-[calc(var(--tab-bar-height)+var(--spacing-player)+env(safe-area-inset-bottom))]"
          : "pb-[calc(var(--tab-bar-height)+env(safe-area-inset-bottom))]",
        className,
      )}
    >
      {children}
    </main>
  )
}
```

Key details:
- `transition-[padding-bottom] duration-200 ease-out` adds smooth padding transition when player appears/disappears (avoids jarring layout shift per research pitfall 1)
- `md:pb-0` disables all mobile padding on desktop (desktop has no tab bar)
- When player inactive: padding = 64px (tab bar) + safe area
- When player active: padding = 64px + 72px (player) + safe area = 136px + safe area
- Footer remains inside the wrapper so it gets pushed above the fixed bars
