    3. Create `src/app/(public)/blog/page.tsx` — Server Component with pagination:
    ```tsx
    import { db } from "@/lib/db"
    import { blogPosts, blogCategories } from "@/db/schema"
    import { eq, desc, and, count } from "drizzle-orm"
    import { PostCard } from "@/components/blog/post-card"
    import { CategoryFilter } from "@/components/blog/category-filter"
    import { ScrollSection } from "@/components/home/scroll-section"
    import { Metadata } from "next"

    const POSTS_PER_PAGE = 9

    export const metadata: Metadata = {
      title: "Blog",
      description: "News, production tips, and behind-the-scenes content from Glitch Studios.",
    }

    export default async function BlogPage({
      searchParams,
    }: {
      searchParams: Promise<{ page?: string; category?: string }>
    }) {
      const params = await searchParams
      const page = Math.max(1, parseInt(params.page || "1", 10))
      const categorySlug = params.category || null

      // Fetch categories
      const categories = await db.select().from(blogCategories)

      // Build query conditions
      const conditions = [eq(blogPosts.status, "published")]
      if (categorySlug) {
        const cat = categories.find(c => c.slug === categorySlug)
        if (cat) conditions.push(eq(blogPosts.categoryId, cat.id))
      }

      // Fetch posts with pagination
      const [posts, [{ total }]] = await Promise.all([
        db.select().from(blogPosts)
          .where(and(...conditions))
          .orderBy(desc(blogPosts.publishedAt))
          .limit(POSTS_PER_PAGE)
          .offset((page - 1) * POSTS_PER_PAGE),
        db.select({ total: count() }).from(blogPosts)
          .where(and(...conditions)),
      ])

      const totalPages = Math.ceil(total / POSTS_PER_PAGE)

      // Empty state
      if (posts.length === 0 && page === 1 && !categorySlug) {
        return (
          <div className="max-w-7xl mx-auto px-4 py-24 text-center">
            <h1 className="font-mono font-bold text-4xl uppercase mb-4 text-white">No posts yet</h1>
            <p className="text-gray-400 text-lg">Check back soon for news, updates, and behind-the-scenes content.</p>
          </div>
        )
      }

      return (
        <div className="max-w-7xl mx-auto px-4 py-16 md:py-24">
          <h1 className="font-mono font-bold text-4xl md:text-5xl uppercase tracking-tight mb-8 text-white">Blog</h1>
          <CategoryFilter categories={categories} activeCategory={categorySlug} />
          <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6 mt-8">
            {posts.map((post, i) => (
              <ScrollSection key={post.id} delay={i * 0.05}>
                <PostCard post={post} />
              </ScrollSection>
            ))}
          </div>
          {/* Pagination */}
          {totalPages > 1 && (
            <nav className="flex justify-center gap-2 mt-12">
              {Array.from({ length: totalPages }, (_, i) => i + 1).map(p => (
                <a
                  key={p}
                  href={`/blog?page=${p}${categorySlug ? `&category=${categorySlug}` : ""}`}
                  className={`px-4 py-2 rounded font-mono text-sm ${
                    p === page
                      ? "bg-gray-800 text-white border border-gray-600"
                      : "text-gray-400 hover:text-white hover:bg-gray-800/50"
                  }`}
                >
                  {p}
                </a>
              ))}
            </nav>
          )}
        </div>
      )
    }
    ```

    1. Create `src/actions/contact.ts` — Server action:
    ```typescript
    "use server"
    import { db } from "@/lib/db"
    import { contactSubmissions } from "@/db/schema"
    import { z } from "zod"

    const contactSchema = z.object({
      name: z.string().min(2, "Name must be at least 2 characters"),
      email: z.string().email("Please enter a valid email address"),
      serviceInterest: z.string().optional(),
      message: z.string().min(10, "Message must be at least 10 characters"),
    })

    export async function submitContactForm(formData: FormData) {
      const raw = {
        name: formData.get("name") as string,
        email: formData.get("email") as string,
        serviceInterest: formData.get("serviceInterest") as string | null,
        message: formData.get("message") as string,
      }

      const result = contactSchema.safeParse(raw)
      if (!result.success) {
        return { success: false, message: "Please check your details and try again.", errors: result.error.flatten().fieldErrors }
      }

      try {
        await db.insert(contactSubmissions).values({
          name: result.data.name,
          email: result.data.email,
          serviceInterest: result.data.serviceInterest || null,
          message: result.data.message,
        })
        return { success: true, message: "Message sent. We will get back to you within 24 hours." }
      } catch (error) {
        return { success: false, message: "Something went wrong. Please try again or email us directly." }
      }
    }
    ```

    import { db } from "@/lib/db"
    import { services } from "@/db/schema"
    import { eq, asc } from "drizzle-orm"
