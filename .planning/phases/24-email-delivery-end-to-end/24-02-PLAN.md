       const EMAIL_FROM = "Glitch Studios <noreply@glitchstudios.io>"

       async function sendBookingModificationEmail(params: {
         to: string
         name: string
         bookingId: string
         service: string
         oldDate: string
         newDate: string | null
         reason?: string
       }) {
         try {
           const { error } = await resend.emails.send({
             from: EMAIL_FROM,
             to: params.to,
             subject: params.newDate === null
               ? "Your Glitch Studios booking was cancelled"
               : "Your Glitch Studios booking was rescheduled",
             react: BookingModificationEmail({
               name: params.name,
               bookingId: params.bookingId,
               service: params.service,
               oldDate: params.oldDate,
               newDate: params.newDate,
               reason: params.reason,
             }),
           })
           if (error) console.error("[email:booking-mod] Resend failed:", error)
         } catch (err) {
           console.error("[email:booking-mod] Unexpected:", err)
         }
       }
