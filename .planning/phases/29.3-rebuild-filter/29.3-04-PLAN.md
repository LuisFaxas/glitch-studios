```ts
import { test, expect } from "@playwright/test"

/**
 * Phase 29.3 cross-engine crash-repro test.
 *
 * Cannot reproduce the actual macOS Metal crash (headless Linux clean), but
 * catches the regression class: listener leaks, DOM leaks, console error
 * storms, page-exception throws.
 *
 * Run:
 *   pnpm exec playwright test tests/29.3-filter-chip-crash.spec.ts --project=webkit
 *   pnpm exec playwright test tests/29.3-filter-chip-crash.spec.ts --project=firefox
 *
 * Both must exit 0.
 *
 * Selector contract:
 *   - [data-testid="leaderboard-filters"] — filter bar root (added in Plan 29.3-03)
 *   - [data-facet-dropdown="Year"]        — Year facet trigger button
 *   - inside the open Year panel: <button aria-pressed="..."> per option chip
 */

const ROUTE = "/tech/rankings/laptops"
const CLICK_BATCH = 20

test("filter chip click does not leak DOM, listeners, or fire console errors (Year facet)", async ({
  page,
}) => {
  const consoleErrors: string[] = []
  const pageExceptions: string[] = []

  page.on("console", (msg) => {
    if (msg.type() === "error") {
      consoleErrors.push(msg.text())
    }
  })
  page.on("pageerror", (err) => {
    pageExceptions.push(err.message)
  })

  // 1. Navigate
  await page.goto(ROUTE)

  // 2. Wait for table to render — at least one row must be present.
  // The wrapper div uses `data-leaderboard-table` (existing selector).
  await page.locator("[data-leaderboard-table]").waitFor({ state: "visible" })
  const rowsBefore = await page
    .locator("[data-leaderboard-table] tbody tr")
    .count()
  expect(rowsBefore).toBeGreaterThan(0)

  // 3. Locate filter root (selector contract from Plan 29.3-03)
  const filterRoot = page.locator('[data-testid="leaderboard-filters"]')
  await expect(filterRoot).toBeVisible()

  // 4. Open the Year dropdown by clicking its trigger
  const yearTrigger = page.locator('[data-facet-dropdown="Year"]')
  await expect(yearTrigger).toBeVisible()
  await yearTrigger.click()

  // The open Year panel renders inside a `role="menu"` (CustomDropdown).
  const yearPanel = page.locator('[data-facet-dropdown="Year"] + [role="menu"]')
  // CustomDropdown renders the menu as a sibling of the trigger inside the
  // wrapper. If the sibling selector doesn't match (depends on relative DOM),
  // fall back to a more general menu near the trigger:
  const yearMenu = (await yearPanel.count()) > 0
    ? yearPanel
    : page.locator('[data-leaderboard-filters] [role="menu"]').first()
  await expect(yearMenu).toBeVisible()

  // 5. Snapshot DOM node count of the filter subtree (leak baseline)
  const nodesBefore = await filterRoot.locator("*").count()

  // 6. Click a Year option chip 20× rapidly. The first option is the first
  // year (e.g., "2024"). Use the first aria-pressed button inside the menu.
  const firstYearChip = yearMenu.locator("button[aria-pressed]").first()
  await expect(firstYearChip).toBeVisible()
  const initialPressed = await firstYearChip.getAttribute("aria-pressed")

  for (let i = 0; i < CLICK_BATCH; i += 1) {
    await firstYearChip.click()
  }

  // 7. Press Escape, expect panel to close (no waitForTimeout — wait on the
  // assertion. anti-pattern guard).
  await page.keyboard.press("Escape")
  await expect(yearMenu).not.toBeVisible({ timeout: 2000 })

  // 8. Snapshot final DOM node count
  const nodesAfter = await filterRoot.locator("*").count()

  // ----- Assertions -----

  // No console errors during the batch
  expect(consoleErrors, `console errors during batch:\n${consoleErrors.join("\n")}`).toEqual([])

  // No page exceptions
  expect(pageExceptions, `page exceptions during batch:\n${pageExceptions.join("\n")}`).toEqual([])

  // Document still interactive after the batch
  await expect(page.locator("body")).toBeVisible()

  // aria-pressed cycles correctly: 20 toggles → ends at the same state it started
  // (20 is even — pressed/unpressed should match initial). Re-open the menu to
  // re-locate the chip (Escape closed it).
  await yearTrigger.click()
  const reopenedMenu = (await yearPanel.count()) > 0
    ? yearPanel
    : page.locator('[data-leaderboard-filters] [role="menu"]').first()
  await expect(reopenedMenu).toBeVisible()
  const finalPressed = await reopenedMenu
    .locator("button[aria-pressed]")
    .first()
    .getAttribute("aria-pressed")
  expect(finalPressed).toBe(initialPressed)
  await page.keyboard.press("Escape")

  // Row count differs from baseline when chip is active. Toggle ON once.
  // (Currently chip is at initial state; click once to activate.)
  await yearTrigger.click()
  const reopenedMenu2 = (await yearPanel.count()) > 0
    ? yearPanel
    : page.locator('[data-leaderboard-filters] [role="menu"]').first()
  await reopenedMenu2.locator("button[aria-pressed]").first().click()
  await page.keyboard.press("Escape")
  const rowsFiltered = await page
    .locator("[data-leaderboard-table] tbody tr")
    .count()
  // The seed corpus has multiple years, so a single-year filter should reduce
  // row count below the unfiltered total (or at most equal). We assert ≤.
  expect(rowsFiltered).toBeLessThanOrEqual(rowsBefore)

  // DOM node delta within ±2 of snapshot (leak guard).
  // Re-snapshot AFTER closing the panel and after the filter has applied;
  // we expect the filter root subtree size to be stable.
  const nodesFinal = await filterRoot.locator("*").count()
  expect(Math.abs(nodesFinal - nodesBefore)).toBeLessThanOrEqual(2)

  // Echo for log readability
  console.log(
    `nodesBefore=${nodesBefore} nodesAfter=${nodesAfter} nodesFinal=${nodesFinal} rowsBefore=${rowsBefore} rowsFiltered=${rowsFiltered}`,
  )
})
```

Notes:
- Selector for the menu: CustomDropdown (leaderboard-filter-sidebar.tsx:103-115) renders the menu inside the same wrapper div as the trigger, so `[data-facet-dropdown="Year"] + [role="menu"]` targets the immediate sibling. If the DOM layout makes the sibling selector unreliable, the fallback `[data-leaderboard-filters] [role="menu"]` is used. Both selectors are written defensively.
- No `page.waitForTimeout` — only assertion-driven waits (`waitFor`, `expect(...).toBeVisible/not.toBeVisible({ timeout })`). This is the anti-pattern guard.
- baseURL `http://localhost:3004` is set in playwright.config — relative `goto(ROUTE)` resolves to that.
- The user runs `pnpm dev` against port 3004 separately before invoking the test (the existing test convention).
