    **STEP 2 — Create `tests/29.3-filter-chip-crash.spec.ts`.**

    ```ts
    /**
     * Phase 29.3 — Filter Chip Crash Architectural Insurance
     *
     * This test cannot reproduce the original macOS GPU crash on headless Linux.
     * Per the debug investigation (.planning/debug/filter-chip-crash-mac-browsers.md),
     * the bug was a Metal/WebRender compositor pressure cascade triggered by:
     *   1. 7 unoptimized 1200×800 PNGs decoded per chip click
     *   2. Permanent filter: drop-shadow GPU layer on the always-mounted LogoTile
     * Both were eliminated as the atomic fix. The PHASE 29.3 rebuild adds
     * architectural insurance: bounded DOM mutations, no third-party popover
     * state, no portals, no listener leaks. THIS TEST verifies that insurance
     * holds — if anyone reintroduces a portal-on-every-render, a listener leak,
     * a node-mounting animation, or a startTransition wedge in the chip path,
     * one of these assertions will fail.
     *
     * Real macOS Safari + Firefox verification is Plan 05 (human gate).
     */
    import { test, expect, type ConsoleMessage } from "@playwright/test"

    const PAGE_URL = "/tech/rankings/laptops"
    const CLICK_COUNT = 20
    const NODE_COUNT_TOLERANCE = 2

    test.describe("29.3 — Filter chip crash architectural insurance", () => {
      test.setTimeout(60_000)

      test("20× chip toggle batch leaves the page interactive with no DOM/listener leak", async ({
        page,
        browserName,
      }) => {
        const consoleErrors: string[] = []
        const pageErrors: string[] = []
        page.on("console", (msg: ConsoleMessage) => {
          if (msg.type() === "error") {
            consoleErrors.push(`[${browserName}] ${msg.text()}`)
          }
        })
        page.on("pageerror", (err) => {
          pageErrors.push(`[${browserName}] ${err.message}`)
        })

        await page.goto(PAGE_URL, { waitUntil: "domcontentloaded" })

        // Wait for the leaderboard table to render at least one row.
        const table = page.locator("[data-leaderboard-table] table, [data-leaderboard-table-mobile] table").first()
        await expect(table).toBeVisible({ timeout: 10_000 })
        const initialRowCount = await table.locator("tbody tr").count()
        expect(initialRowCount).toBeGreaterThan(0)

        // Confirm the rebuilt filter subtree is mounted.
        const filterRoot = page.locator('[data-testid="leaderboard-filters"]').first()
        await expect(filterRoot).toBeVisible({ timeout: 5_000 })

        // Snapshot the filter subtree node count BEFORE any interaction.
        const initialNodeCount = await filterRoot.evaluate(
          (el) => el.getElementsByTagName("*").length,
        )

        // Open the Year dropdown.
        const yearTrigger = page.locator('[data-filter-dropdown-trigger="Year"]').first()
        await expect(yearTrigger).toBeVisible({ timeout: 5_000 })
        await yearTrigger.click()

        // Locate the open panel and the first chip inside it.
        const yearPanel = page.locator('[data-filter-dropdown-panel="Year"]').first()
        await expect(yearPanel).toBeVisible({ timeout: 3_000 })
        const firstChip = yearPanel.locator("button[aria-pressed]").first()
        await expect(firstChip).toBeVisible({ timeout: 3_000 })

        // Record initial chip state and table row count.
        const initialPressed = await firstChip.getAttribute("aria-pressed")
        expect(initialPressed).toBe("false")

        // Click the chip CLICK_COUNT times. Track the table row counts seen
        // during the batch — at least one must differ from initialRowCount.
        const rowCountsSeen = new Set<number>()
        rowCountsSeen.add(initialRowCount)
        for (let i = 0; i < CLICK_COUNT; i++) {
          await firstChip.click()
          // Brief pause to let React commit; no Promise resolution machinery beyond this.
          await page.waitForTimeout(50)
          const rc = await table.locator("tbody tr").count()
          rowCountsSeen.add(rc)
        }

        // ASSERT 1+2: No console errors, no page exceptions during the batch.
        expect(consoleErrors, `console errors on ${browserName}: ${consoleErrors.join("\\n")}`).toHaveLength(0)
        expect(pageErrors, `page exceptions on ${browserName}: ${pageErrors.join("\\n")}`).toHaveLength(0)

        // ASSERT 3: document is still complete.
        const readyState = await page.evaluate(() => document.readyState)
        expect(readyState).toBe("complete")

        // ASSERT 4: Page is still responsive (eval round-trip in <2s).
        const t0 = Date.now()
        const evalResult = await page.evaluate(() => 1 + 1)
        const evalMs = Date.now() - t0
        expect(evalResult).toBe(2)
        expect(evalMs, `responsiveness probe took ${evalMs}ms on ${browserName}`).toBeLessThan(2000)

        // ASSERT 5: 20 toggles → chip aria-pressed returns to initial value.
        const finalPressed = await firstChip.getAttribute("aria-pressed")
        expect(finalPressed).toBe(initialPressed)

        // ASSERT 6: At least one intermediate row count differed from initial
        // (proves the filter actually filters; not just a static no-op).
        expect(
          rowCountsSeen.size,
          `row counts seen during batch: ${Array.from(rowCountsSeen).join(",")}`,
        ).toBeGreaterThan(1)

        // ASSERT 7: DOM-leak guard. Filter subtree node count within ±tolerance.
        // Close the panel first so the comparison is apples-to-apples (panel was
        // closed when initialNodeCount was snapshotted). Use a locator-based
        // wait — NOT page.waitForTimeout — so the test waits exactly as long as
        // needed for the panel unmount to commit. Webkit's event loop under
        // heavy click load can be slower than 100ms; a fixed timeout produces
        // false-positive leak signals.
        await page.keyboard.press("Escape")
        await expect(yearPanel).not.toBeVisible({ timeout: 2000 })
        const finalNodeCount = await filterRoot.evaluate(
          (el) => el.getElementsByTagName("*").length,
        )
        const delta = Math.abs(finalNodeCount - initialNodeCount)
        expect(
          delta,
          `DOM-leak guard: filter subtree grew from ${initialNodeCount} to ${finalNodeCount} (delta ${delta}) on ${browserName}; tolerance ±${NODE_COUNT_TOLERANCE}`,
        ).toBeLessThanOrEqual(NODE_COUNT_TOLERANCE)
      })
    })
    ```

    Implementation rules:
    - Mirror the harness shape from `tests/forensics-overlay-leak.spec.ts`: `page.evaluate` for DOM measurements, no MutationObservers (the spec's own machinery would skew node counts).
    - Use a 50ms `waitForTimeout` between clicks. Shorter is too aggressive for Webkit's event loop; longer obscures the leak signal.
    - The DOM-leak tolerance is ±2 to allow for legitimate React reconciliation noise (e.g. a transient `<span>` for a count-badge update). The original CONTEXT.md spec said ±2 — match it exactly.
    - The rowCountsSeen set assertion is the surrogate for "the filter actually filters". A definitive expected count requires fixture-aware assertions; using "at least one intermediate count differed from initial" is the engine-agnostic surrogate that proves the filter wires through.
    - The Escape-press + locator-based `not.toBeVisible({ timeout: 2000 })` pattern (final assertion) ensures the panel is closed at both snapshot points so the comparison is fair, AND avoids false-positive leak signals on slower webkit/firefox event loops where a fixed 100ms timeout is insufficient.
    - The test runs against the dev server already running on the standard Playwright config baseURL (`http://localhost:3004`).
    - DO NOT introduce a `test.use({ project: ... })` filter. The test must run on every project the config defines; the acceptance criteria invocations below force webkit and firefox runs explicitly.
    - DO NOT add a MutationObserver to the test — it would itself be a listener that affects the DOM-leak baseline.
