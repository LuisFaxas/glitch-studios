    ```tsx
    "use client"
    import {
      useCallback,
      useEffect,
      useRef,
      useState,
      type KeyboardEvent as ReactKeyboardEvent,
    } from "react"
    import { ChevronDown } from "lucide-react"

    export interface FilterDropdownProps {
      label: string
      activeCount?: number
      onClose?: () => void
      panelClassName?: string
      align?: "left" | "right"
      children: React.ReactNode
    }

    export function FilterDropdown({
      label,
      activeCount = 0,
      onClose,
      panelClassName,
      align = "left",
      children,
    }: FilterDropdownProps) {
      const [open, setOpen] = useState(false)
      const triggerRef = useRef<HTMLButtonElement | null>(null)
      const panelRef = useRef<HTMLDivElement | null>(null)
      // Tracks whether the most recent close was triggered by Escape so we
      // can return focus only on keyboard close (mouse-outside should not
      // steal focus from wherever the user clicked).
      const lastCloseViaKeyboard = useRef(false)
      // Latch onClose into a ref so we don't re-attach listeners when the
      // caller passes a fresh inline callback every render.
      const onCloseRef = useRef(onClose)
      useEffect(() => {
        onCloseRef.current = onClose
      }, [onClose])

      // Listener lifecycle: attach on open, detach on close. No portals.
      useEffect(() => {
        if (!open) return
        function handleMouseDown(e: MouseEvent) {
          const target = e.target as Node | null
          if (!target) return
          if (panelRef.current?.contains(target)) return
          if (triggerRef.current?.contains(target)) return
          lastCloseViaKeyboard.current = false
          setOpen(false)
        }
        function handleKeyDown(e: globalThis.KeyboardEvent) {
          if (e.key === "Escape") {
            lastCloseViaKeyboard.current = true
            setOpen(false)
          }
        }
        document.addEventListener("mousedown", handleMouseDown)
        document.addEventListener("keydown", handleKeyDown)
        return () => {
          document.removeEventListener("mousedown", handleMouseDown)
          document.removeEventListener("keydown", handleKeyDown)
        }
      }, [open])

      // Fire onClose on the open->closed transition, and return focus on
      // keyboard-driven close.
      const previouslyOpen = useRef(false)
      useEffect(() => {
        if (previouslyOpen.current && !open) {
          if (lastCloseViaKeyboard.current) {
            triggerRef.current?.focus()
          }
          onCloseRef.current?.()
        }
        previouslyOpen.current = open
      }, [open])

      const handleTriggerClick = useCallback(() => {
        setOpen((prev) => !prev)
      }, [])

      const handleTriggerKeyDown = useCallback(
        (e: ReactKeyboardEvent<HTMLButtonElement>) => {
          // Allow Enter/Space to toggle; let the browser handle Tab.
          if (e.key === "Enter" || e.key === " ") {
            e.preventDefault()
            setOpen((prev) => !prev)
          }
        },
        [],
      )

      const triggerLabel = activeCount > 0 ? `${label} (${activeCount})` : label
      const isActive = activeCount > 0
      const panelAlignClass = align === "right" ? "right-0" : "left-0"

      return (
        <div className="relative inline-block">
          <button
            ref={triggerRef}
            type="button"
            onClick={handleTriggerClick}
            onKeyDown={handleTriggerKeyDown}
            aria-expanded={open}
            aria-haspopup="listbox"
            data-filter-dropdown-trigger={label}
            className={
              "inline-flex items-center gap-1.5 border px-3 py-1.5 font-mono text-[11px] uppercase tracking-wide transition-colors " +
              (isActive
                ? "border-[#f5f5f0] bg-[#1a1a1a] text-[#f5f5f0]"
                : "border-[#222] bg-[#0a0a0a] text-[#888] hover:border-[#444] hover:text-[#ccc]")
            }
          >
            <span>{triggerLabel}</span>
            <ChevronDown className="h-3 w-3" aria-hidden />
          </button>
          {open && (
            <div
              ref={panelRef}
              role="listbox"
              data-filter-dropdown-panel={label}
              className={
                (panelClassName ??
                  "absolute top-[calc(100%+8px)] z-50 min-w-[200px] max-w-[320px] border border-[#222] bg-[#0a0a0a] p-3") +
                " " +
                panelAlignClass
              }
            >
              {children}
            </div>
          )}
        </div>
      )
    }

    export default FilterDropdown
    ```

    Implementation rules (all locked by CONTEXT.md `<decisions>`):
    - NO `import` from `@base-ui-components`, `@radix-ui/*`, `@headlessui`, `framer-motion`, `motion`, `nuqs`, `next/navigation`. Only React + lucide-react are allowed.
    - NO `createPortal`, NO `ReactDOM.createPortal`. The panel renders as a sibling DOM node within the trigger's parent.
    - NO transform animation, NO `transition` other than the existing `transition-colors` class on the trigger (color hover state only — same as Phase 29.1 chip styling).
    - The panel uses CONDITIONAL rendering (`{open && <div>…}`), NEVER `display: none`. This is the bounded-DOM-mutations rule from CONTEXT.md.
    - Listeners attach inside the `useEffect` only when `open === true` and detach in the cleanup. The early `if (!open) return` guarantees no listener exists while closed.
    - The `onClose` callback is wrapped in a ref so caller-passed inline functions don't re-trigger the open/close effect.
    - The `data-filter-dropdown-trigger` and `data-filter-dropdown-panel` attributes carry the label so Plan 04's Playwright test can target specific facets without guessing accessible names.
    - Visual style mirrors the existing `FilterFacetDropdown` trigger in `leaderboard-filter-sidebar.tsx` lines 178-194 (same border colors, padding, font-mono uppercase tracking) so the visual contract from Phase 29.1 is preserved.
