IMPORTANT: The upload flow must NOT send file data to the server action. The server action only handles metadata and presigned URL generation. The client uploads directly to R2. This addresses the HIGH review concern about `uploadMedia(formData)` defeating R2 direct upload.

```typescript
// Step 1: Client calls this to get a presigned upload URL (metadata only, no file data)
export async function getMediaUploadUrl(data: {
  filename: string
  mimeType: string
  size: number
}): Promise<{ uploadUrl: string; key: string }> {
  await requirePermission("manage_media")
  const slugifiedName = data.filename.replace(/[^a-z0-9.]/gi, "-").toLowerCase()
  const key = `media/${Date.now()}-${slugifiedName}`
  const uploadUrl = await getUploadUrl(key, data.mimeType)
  return { uploadUrl, key }
}

// Step 2: After client uploads file directly to R2, call this to record in DB
export async function confirmMediaUpload(data: {
  key: string
  filename: string
  mimeType: string
  size: number
  width?: number
  height?: number
  duration?: number
  alt?: string
}): Promise<MediaAsset> {
  await requirePermission("manage_media")
  const url = getPublicUrl(data.key)
  const [asset] = await db.insert(mediaAssets).values({
    key: data.key,
    filename: data.filename,
    url,
    mimeType: data.mimeType,
    size: data.size,
    width: data.width,
    height: data.height,
    duration: data.duration,
    alt: data.alt,
  }).returning()
  return asset
}

// Delete -- warns user but does not block (v1 has no usage tracking)
export async function deleteMedia(mediaId: string): Promise<void> {
  await requirePermission("manage_media")
  const [asset] = await db.select().from(mediaAssets).where(eq(mediaAssets.id, mediaId))
  if (asset) {
    // Delete from R2
    await r2Client.send(new DeleteObjectCommand({
      Bucket: process.env.R2_BUCKET_NAME!,
      Key: asset.key,
    }))
    // Delete from DB
    await db.delete(mediaAssets).where(eq(mediaAssets.id, mediaId))
  }
}
```
