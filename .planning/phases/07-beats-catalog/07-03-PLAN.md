import { test, expect } from "@playwright/test"

const BASE_URL = process.env.BASE_URL || "https://glitch-studios.codebox.local"

test.describe("Beats Catalog Visual Verification", () => {
  test("desktop card view (1440px)", async ({ page }) => {
    await page.setViewportSize({ width: 1440, height: 900 })
    await page.goto(`${BASE_URL}/beats`)
    await page.waitForLoadState("networkidle")
    // Verify filter bar is visible
    await expect(page.locator('input[placeholder="Search beats..."]')).toBeVisible()
    // Verify at least one beat card is visible (if data exists)
    await page.screenshot({
      path: ".planning/audit-screenshots/07-beats-card-1440.png",
      fullPage: true,
    })
  })

  test("desktop list view (1440px)", async ({ page }) => {
    await page.setViewportSize({ width: 1440, height: 900 })
    await page.goto(`${BASE_URL}/beats?view=list`)
    await page.waitForLoadState("networkidle")
    await page.screenshot({
      path: ".planning/audit-screenshots/07-beats-list-1440.png",
      fullPage: true,
    })
  })

  test("mobile card view (375px)", async ({ page }) => {
    await page.setViewportSize({ width: 375, height: 812 })
    await page.goto(`${BASE_URL}/beats`)
    await page.waitForLoadState("networkidle")
    await page.screenshot({
      path: ".planning/audit-screenshots/07-beats-card-375.png",
      fullPage: true,
    })
  })

  test("mobile list view (375px)", async ({ page }) => {
    await page.setViewportSize({ width: 375, height: 812 })
    await page.goto(`${BASE_URL}/beats?view=list`)
    await page.waitForLoadState("networkidle")
    await page.screenshot({
      path: ".planning/audit-screenshots/07-beats-list-375.png",
      fullPage: true,
    })
  })

  test("filtered view (1440px)", async ({ page }) => {
    await page.setViewportSize({ width: 1440, height: 900 })
    await page.goto(`${BASE_URL}/beats?genre=Hip-Hop`)
    await page.waitForLoadState("networkidle")
    await page.screenshot({
      path: ".planning/audit-screenshots/07-beats-filtered-1440.png",
      fullPage: true,
    })
