**handlePlay function** (copy pattern from beat-row.tsx):
```typescript
function handlePlay(e: React.MouseEvent) {
  e.stopPropagation()
  if (!beat.previewAudioUrl) return
  if (isActivePlaying) { pause() }
  else { play({ id: beat.id, title: beat.title, artist: beat.producers[0]?.name ?? "Glitch Studios", previewAudioUrl: beat.previewAudioUrl, coverArtUrl: beat.coverArtUrl }) }
}
```
