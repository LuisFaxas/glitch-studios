<objective>
Refactor the two pre-ingest queries in `src/lib/tech/queries.ts` so `/tech/compare` and the tech homepage benchmark spotlight survive Phase 16 ingest without regression.

Purpose:
- `getBenchmarkRunsForProducts` currently returns every historical run per (product, test) pair. After Phase 16 ingests even one re-run, the `/tech/compare` tool would show 2+ cells per metric per product (broken). Fix with `DISTINCT ON (productId, testId)` ordered `recordedAt DESC`, filtered `superseded = false` and `mode = 'ac'` (compare is AC-only per ARCHITECTURE.md).

    ```typescript
    export async function getBenchmarkRunsForProducts(
      productIds: string[],
    ): Promise<PublicBenchmarkRun[]> {
      if (productIds.length === 0) return []

      // DISTINCT ON (product_id, test_id) filtered to superseded=false + mode='ac' ordered by
      // recorded_at DESC (newest canonical run wins).
      //
      // /tech/compare is AC-only by editorial convention (ARCHITECTURE.md §compare). Filtering
      // to mode='ac' at the query keeps the PublicBenchmarkRun shape unchanged for consumers
      // (no `mode` field leaked into the UI).
      //
      // Postgres constraint: DISTINCT ON cols MUST lead ORDER BY in same order; the tie-breaker
      // (recordedAt DESC) is appended after. Re-ordering breaks with:
      //   "SELECT DISTINCT ON expressions must match initial ORDER BY expressions"
      const rows = await db
        .selectDistinctOn(
          [techBenchmarkRuns.productId, techBenchmarkRuns.testId],
          {
            productId: techBenchmarkRuns.productId,
            testId: techBenchmarkTests.id,
            testName: techBenchmarkTests.name,
            unit: techBenchmarkTests.unit,
            direction: techBenchmarkTests.direction,
            score: techBenchmarkRuns.score,
            recordedAt: techBenchmarkRuns.recordedAt,
            sortOrder: techBenchmarkTests.sortOrder,
          },
        )
        .from(techBenchmarkRuns)
        .innerJoin(
          techBenchmarkTests,
          eq(techBenchmarkRuns.testId, techBenchmarkTests.id),
        )
        .where(and(
          inArray(techBenchmarkRuns.productId, productIds),
          eq(techBenchmarkRuns.superseded, false),
          eq(techBenchmarkRuns.mode, "ac"),
        ))
        .orderBy(
          asc(techBenchmarkRuns.productId),
          asc(techBenchmarkRuns.testId),
          desc(techBenchmarkRuns.recordedAt),
        )

      // Post-sort by display order. DISTINCT ON requires the specific ORDER BY above,
      // so re-sort in memory for presentation.
      const sorted = rows.sort((a, b) => {
        if (a.sortOrder !== b.sortOrder) return a.sortOrder - b.sortOrder
        return a.testName.localeCompare(b.testName)
      })

      return sorted.map((r) => ({
        productId: r.productId,
        testId: r.testId,
        testName: r.testName,
        unit: r.unit,
        direction: r.direction,
        score: Number(r.score),
        recordedAt: r.recordedAt,
      }))
    }
    ```

    **Key decisions baked in (from CONTEXT D-16 + RESEARCH recommendation b):**
    1. DISTINCT ON is `(productId, testId)` — NOT `(productId, testId, mode)` — because we filter to `mode='ac'` first. This keeps the public shape backward-compatible.
    2. Filter `superseded = false` — live rows only.
    3. In-memory re-sort by `(sortOrder, testName)` preserves existing display order in compare UI (original `orderBy(asc(sortOrder), asc(name))` semantics).
    4. PublicBenchmarkRun interface shape is UNCHANGED — no new `mode` field. Consumers (`/tech/compare`) see no change.

    **Do NOT:** add a `mode` field to `PublicBenchmarkRun`. Do NOT change the function signature. Do NOT rename the function.

    export async function getBenchmarkSpotlight(): Promise<BenchmarkSpotlight | null> {
      // Resolve the spotlight test via the rubric-map canonical key (D-17).
      // Renaming the test row in tech_benchmark_tests no longer breaks the spotlight —
      // the rubric-map constant is the source of truth.
      const spotlightEntry = RUBRIC_V1_1["cpu:geekbench6:multi"]
      if (!spotlightEntry) return null

      // Look up the test id by natural key (discipline, mode, name) — unique per v1.1 seed.
      const [spotlightTest] = await db
        .select({ id: techBenchmarkTests.id })
        .from(techBenchmarkTests)
        .where(and(
          eq(techBenchmarkTests.discipline, spotlightEntry.discipline),
          eq(techBenchmarkTests.mode, spotlightEntry.mode),
          eq(techBenchmarkTests.name, spotlightEntry.name),
        ))
        .limit(1)

      if (!spotlightTest) return null

      // Top AC run for this test, scoped to products with published reviews.
      const [candidate] = await db
        .select({
          productId: techBenchmarkRuns.productId,
          score: techBenchmarkRuns.score,
        })
        .from(techBenchmarkRuns)
        .innerJoin(techReviews, and(
          eq(techReviews.productId, techBenchmarkRuns.productId),
          eq(techReviews.status, "published"),
        ))
        .where(and(
          eq(techBenchmarkRuns.testId, spotlightTest.id),
          eq(techBenchmarkRuns.mode, "ac"),
          eq(techBenchmarkRuns.superseded, false),
        ))
        .orderBy(desc(techBenchmarkRuns.score))
        .limit(1)

      if (!candidate) return null
      const { productId } = candidate

      const productRow = await db
        .select({
          product: techProducts,
          heroImageUrl: mediaAssets.url,

        .leftJoin(mediaAssets, eq(techProducts.heroImageId, mediaAssets.id))
        .where(eq(techProducts.id, productId))
        .limit(1)

      if (productRow.length === 0) return null
      const p = productRow[0]

      // Top 2 other AC runs for this product (keeps existing behavior).

     * Phase 15 runtime regression check.
     *
     * Asserts that getBenchmarkRunsForProducts returns NO duplicate (productId, testId) pairs
     * after the DISTINCT ON refactor, and that getBenchmarkSpotlight resolves via id lookup
     * (not ilike). Run after any rubric seed + benchmark-run inserts during Phase 15/16 dev.

     * Exit code: 0 = OK, 1 = regression detected.
     * Safe to delete after Phase 16 adds its own ingest tests.
     */
    import "dotenv/config"
    import { db } from "../client"
    import { techProducts } from "../schema"
    import {
      getBenchmarkRunsForProducts,
      getBenchmarkSpotlight,
    } from "../../lib/tech/queries"

    async function main() {
      // Grab up to 5 product ids from the DB — if no products exist, the function
      // should still return [] without error.
      const productSample = await db
        .select({ id: techProducts.id })
        .from(techProducts)
        .limit(5)
      const productIds = productSample.map((p) => p.id)

      // Always exercise the empty-input path first.
      const empty = await getBenchmarkRunsForProducts([])
      if (empty.length !== 0) {
        console.error("FAIL: getBenchmarkRunsForProducts([]) returned non-empty:", empty)
        process.exit(1)
      }

      const rows = await getBenchmarkRunsForProducts(productIds)

      // Check no duplicate (productId, testId) pairs — if any exist, DISTINCT ON is broken.
      const seen = new Map<string, number>()
      for (const r of rows) {
        const key = `${r.productId}::${r.testId}`
        seen.set(key, (seen.get(key) ?? 0) + 1)
      }
      const dupes = [...seen.entries()].filter(([, n]) => n > 1)
      if (dupes.length > 0) {
        console.error("FAIL: duplicate (productId, testId) pairs found:", dupes)
        process.exit(1)
      }

      // Spotlight — if RUBRIC_V1_1 and seed are wired correctly, this resolves OR
      // returns null (null is valid when no published review has a Geekbench run yet).
      const spotlight = await getBenchmarkSpotlight()
      // Not asserting a specific value — just that it does not throw.

      console.log(
        `OK: ${rows.length} canonical rows across ${productIds.length} products, no duplicate pairs. ` +
        `Spotlight: ${spotlight ? spotlight.productName : "null (no matching published review yet)"}`
      )
      process.exit(0)
    }

    main().catch((e) => {
      console.error(e)
      process.exit(1)
    })
