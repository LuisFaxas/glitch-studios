    ```ts
    import { test, expect } from "@playwright/test"

    /**
     * 29.2-01 — TechHero size variants
     * Asserts that the three height variants render at the correct pixel heights.
     * Uses the [data-tech-hero] + [data-size] hooks for stable locating.
     *
     * NOTE: The spec tests the default size on /tech/rankings (existing page with
     * TechHero already mounted). Plans 29.2-02 through 29.2-10 will add TechHero
     * to the remaining surfaces. The tall variant test mounts a TechHero with
     * size="tall" on /tech/about once Plan 29.2-02 ships — see comment below.
     *
     * For this plan (29.2-01), we verify:
     *   - The existing /tech/rankings hero still renders at 280px (no regression)
     *   - The data-size attribute reflects the correct value when size is passed
     *
     * Because no page in the codebase yet renders size="tall" or size="compact",
     * we verify height-map coverage via TypeScript only (pnpm tsc --noEmit) and
     * by reading the component's rendered inline style in a targeted nav to /tech/rankings.
     * Full tall-variant visual assertion is in 29.2-02-PLAN's spec.
     */
    test.describe("29.2-01 — TechHero size variants (no-regression + attribute check)", () => {
      test.setTimeout(30_000)

      test("default size: /tech/rankings hero renders at 280px", async ({ page }) => {
        await page.goto("/tech/rankings")
        const hero = page.locator("[data-tech-hero]").first()
        await expect(hero).toBeVisible({ timeout: 5_000 })

        // The inner gradient div (second child of data-tech-hero) has the height class.
        // We check the computed offsetHeight is >= 280 to allow for sub-pixel rendering.
        const innerDiv = hero.locator("div").first()
        const height = await innerDiv.evaluate((el) => (el as HTMLElement).offsetHeight)
        expect(height).toBeGreaterThanOrEqual(280)
        expect(height).toBeLessThan(300) // still "default", not tall
      })

      test("default size: /tech/rankings/laptops hero renders at 280px", async ({ page }) => {
        await page.goto("/tech/rankings/laptops")
        const hero = page.locator("[data-tech-hero]").first()
        await expect(hero).toBeVisible({ timeout: 5_000 })
        const innerDiv = hero.locator("div").first()
        const height = await innerDiv.evaluate((el) => (el as HTMLElement).offsetHeight)
        expect(height).toBeGreaterThanOrEqual(280)
        expect(height).toBeLessThan(300)
      })

      test("data-tech-hero and data-tone attributes still present after size prop addition", async ({ page }) => {
        await page.goto("/tech/rankings")
        const hero = page.locator("[data-tech-hero]").first()
        await expect(hero).toBeVisible({ timeout: 5_000 })
        await expect(hero).toHaveAttribute("data-tone", "cyan")
      })

      test("no auto-running animation on hero h1 (regression check)", async ({ page }) => {
        await page.goto("/tech/rankings")
        const h1 = page.getByRole("heading", { level: 1, name: /^Rankings$/i })
        await expect(h1).toBeVisible()
        const sample1 = await h1.evaluate((el) => getComputedStyle(el).animationName)
        await page.waitForTimeout(500)
        const sample2 = await h1.evaluate((el) => getComputedStyle(el).animationName)
        expect(sample1).toBe(sample2)
        expect(sample1).toBe("none")
      })
    })
    ```
