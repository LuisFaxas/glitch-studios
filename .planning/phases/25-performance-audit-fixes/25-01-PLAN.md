    ```ts
    import { test, expect, type Page } from "@playwright/test"
    const BASE_URL = process.env.PLAYWRIGHT_BASE_URL || "http://localhost:3004"

    async function loginAsAdmin(page: Page): Promise<boolean> {
      const email = process.env.ADMIN_EMAIL
      const password = process.env.ADMIN_PASSWORD
      if (!email || !password) return false
      const res = await page.request.post(`${BASE_URL}/api/auth/sign-in/email`, {
        data: { email, password },
      })
      return res.ok()
    }

    test.describe("25-01 Admin nav perf", () => {
      test("Studios -> Tech context switch under 1500ms", async ({ page }) => {
        const authed = await loginAsAdmin(page)
        test.skip(!authed, "Admin credentials not available")
        await page.goto(`${BASE_URL}/admin`, { waitUntil: "networkidle" })

        const start = Date.now()
        await page.getByRole("tab", { name: /^Tech$/i }).click()
        await page.waitForURL(`${BASE_URL}/admin/tech`, { timeout: 5_000 })
        // Skeleton should appear basically instantly (< 100ms); use domcontentloaded.
        await page.waitForLoadState("domcontentloaded")
        const elapsed = Date.now() - start

        console.log(`[25-01] Studios -> Tech elapsed: ${elapsed}ms`)
        // Perceived < 500ms is the goal; set test threshold more forgiving (1500ms)
        // since dev mode is slow. Record number for the SUMMARY.
        expect(elapsed).toBeLessThan(1500)
      })

      test("Tech -> Studios context switch under 1500ms", async ({ page }) => {
        const authed = await loginAsAdmin(page)
        test.skip(!authed, "Admin credentials not available")
        await page.goto(`${BASE_URL}/admin/tech`, { waitUntil: "networkidle" })

        const start = Date.now()
        await page.getByRole("tab", { name: /^Studios$/i }).click()
        await page.waitForURL(`${BASE_URL}/admin`, { timeout: 5_000 })
        await page.waitForLoadState("domcontentloaded")
        const elapsed = Date.now() - start

        console.log(`[25-01] Tech -> Studios elapsed: ${elapsed}ms`)
        expect(elapsed).toBeLessThan(1500)
      })
    })
    ```

    Run `pnpm exec playwright test tests/25-01-admin-nav-perf.spec.ts --reporter=line`. Both must pass. Capture the elapsed numbers into SUMMARY.md.
